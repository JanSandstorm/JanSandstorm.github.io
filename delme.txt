  <script>

  const container = document.getElementById("news-list");

  // üßÆ 2Ô∏è‚É£ Filter the posts array for filenames containing the word "news"
  const newsPosts = posts.filter(filename => filename.toLowerCase().includes("news"));

  // üß± 3Ô∏è‚É£ Loop through each matching post and build a list item
  newsPosts.forEach(filename => {
    // Create a new <li> element
    const li = document.createElement("li");
    // Create a clickable link for that file
    const a = document.createElement("a");
    a.href = filename;           // link to the HTML page
    a.textContent = filename.replace(".html", ""); // text shown to user
    // Add the link to the list item
    li.appendChild(a);
    // Add the list item to the container
    container.appendChild(li);
  });
